\newacronym{ALS}{ALS}{Alternating Least Squares — thuật toán tối ưu hóa
    xen kẽ cho Matrix Factorization trên dữ liệu ngầm định}

\newacronym{ANN}{ANN}{Approximate Nearest Neighbors — tìm kiếm láng giềng
    gần đúng hiệu quả cho không gian chiều cao}

\newacronym{API}{API}{Application Programming Interface — Giao diện lập
    trình ứng dụng}

\newacronym{ASGI}{ASGI}{Asynchronous Server Gateway Interface — chuẩn giao
    tiếp máy chủ bất đồng bộ cho Python}

\newacronym{BPR}{BPR}{Bayesian Personalized Ranking — phương pháp huấn
    luyện xếp hạng cá nhân hóa cho dữ liệu ngầm định}

\newacronym{CF}{CF}{Collaborative Filtering — Lọc cộng tác, phương pháp
    gợi ý dựa trên hành vi của cộng đồng người dùng}

\newacronym{CSV}{CSV}{Comma-Separated Values — định dạng file dữ liệu
    phân cách bằng dấu phẩy}

\newacronym{EDA}{EDA}{Exploratory Data Analysis — Phân tích khám phá dữ
    liệu}

\newacronym{GD}{GD}{Gradient Descent — thuật toán tối ưu hóa theo hướng
    gradient giảm dần}

\newacronym{HTML}{HTML}{HyperText Markup Language — Ngôn ngữ đánh dấu
    siêu văn bản, dùng xây dựng giao diện web}

\newacronym{HTTP}{HTTP}{HyperText Transfer Protocol — giao thức truyền
    siêu văn bản, nền tảng của World Wide Web}

\newacronym{JSON}{JSON}{JavaScript Object Notation — định dạng trao đổi
    dữ liệu nhẹ, dễ đọc cho cả người và máy}

\newacronym{MAE}{MAE}{Mean Absolute Error — Sai số tuyệt đối trung bình,
    chỉ số đánh giá độ chính xác dự đoán rating}

\newacronym{MF}{MF}{Matrix Factorization — Phân rã ma trận, kỹ thuật
    Model-based Collaborative Filtering}

\newacronym{MRR}{MRR}{Mean Reciprocal Rank — chỉ số đánh giá chất lượng
    xếp hạng trong hệ thống gợi ý}

\newacronym{NCF}{NCF}{Neural Collaborative Filtering — hệ thống gợi ý
    kết hợp mạng neural với Collaborative Filtering}

\newacronym{NDCG}{NDCG}{Normalized Discounted Cumulative Gain — chỉ số
    đánh giá chất lượng xếp hạng có tính đến vị trí trong danh sách}

\newacronym{RAM}{RAM}{Random Access Memory — bộ nhớ truy cập ngẫu nhiên}

\newacronym{REST}{REST}{Representational State Transfer — kiến trúc thiết
    kế API web theo nguyên tắc không trạng thái}

\newacronym{RMSE}{RMSE}{Root Mean Squared Error — Căn bậc hai của sai số
    bình phương trung bình, chỉ số đánh giá độ chính xác dự đoán rating}

\newacronym{SGD}{SGD}{Stochastic Gradient Descent — Hạ gradient ngẫu
    nhiên, cập nhật tham số sau mỗi mẫu dữ liệu đơn lẻ}

\newacronym{SVD}{SVD}{Singular Value Decomposition — Phân rã giá trị suy
    biến, nền tảng của các thuật toán Matrix Factorization}

\newacronym{UI}{UI}{User Interface — Giao diện người dùng}
